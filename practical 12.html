#include #include #include #include #include using namespace std; struct Question { string ques; vector options; int correct; }; int main() { ifstream file("questions.txt"); ofstream scoreFile("scoreboard.txt", ios::app); vector quiz; string line; // Read file while (getline(file, line)) { Question q; vector parts; string temp = ""; for (char c : line) { if (c == '|') { parts.push_back(temp); temp = ""; } else { temp += c; } } parts.push_back(temp); q.ques = parts[0]; for (int i = 1; i <= 4; i++) { q.options.push_back(parts[i]); } q.correct = stoi(parts[5]) - 1; quiz.push_back(q); } srand(time(0)); int score = 0; // Ask questions for (int i = 0; i < 5; i++) { cout << "\nQ" << i + 1 << ": " << quiz[i].ques << endl; vector> opts; for (int j = 0; j < 4; j++) { opts.push_back({quiz[i].options[j], j}); } random_shuffle(opts.begin(), opts.end()); int correctIndex; for (int j = 0; j < 4; j++) { cout << j + 1 << ". " << opts[j].first << endl; if (opts[j].second == quiz[i].correct) { correctIndex = j; } } int ans; cout << "Enter answer: "; cin >> ans; if (ans - 1 == correctIndex) { cout << "Correct!\n"; score++; } else { cout << "Wrong!\n"; } } cout << "\nYour Score: " << score << "/5\n"; // Save score scoreFile << "Score: " << score << "/5\n"; file.close(); scoreFile.close(); return 0; }
